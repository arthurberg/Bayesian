Constant-risk model 

model= function(){

  ## Likelihood
  for (i in 1:12) {
    y[i] ~ dbin(theta, n[i])
    res[i] <- (y[i] - n[i]*theta)/sqrt(n[i]*theta*(1-theta)) 
    res2[i] <- res[i]*res[i]
    }
  
  ## prior
  theta ~ dunif(0, 1)
  X2.obs <- sum(res2[])

  }


data=list(n = bristol$operations, y=bristol$deaths)

library(R2jags)

Hierarchical model 

model= function(){

  ## Likelihood
  for (i in 1:12) {
    y[i] ~ dbin(theta[i], n[i])
    logit(theta[i]) <- logit.theta[i]
    logit.theta[i] ~ dnorm(mu,inv.sigma.squared)
    }
  
  ## priors
  inv.sigma.squared <- 1/pow(sigma,2)
  sigma ~ dunif(0,100)
  mu ~ dunif(-100,100)
  }


data=list(n = bristol$operations, y=bristol$deaths)
